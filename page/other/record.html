    # 用法

    record() API 用于自定义埋点。

    使用 record() 函数需要在 SDK 引入参数中设置 variable 字段，以设置参数variable=jw_collector 为例，SDK 安装完成后可以通过 window.jw_collector.record() 函数提交采集数据，record() 函数仅接收一个字符串类型的参数。

    # Script 标签引入使用示例

    
// 引入 SDK
<script>
  (function() {
    var scriptElement = document.createElement("script");
    scriptElement.id="jw-trigger"
    scriptElement.src = "http://127.0.0.1:28080/trigger.min.js?type=script&variable=jw_collector&enviroment=TEST&logLevel=DEBUG";
    var existedScript = document.getElementsByTagName("script")[0];
    existedScript.parentNode.insertBefore(scriptElement, existedScript);
  })();
</script>

// 自定义埋点、埋点内容为对象时可先使用 JSON.stringify() 序列化
window.jw_collector.record('自定义埋点字符串');

    # NPM 包引入使用示例

    
import Vue from 'vue'
import App from './App.vue'
import router from './router'

// 引入 SDK
import Trigger from 'jw-trigger'
import {
  SdkEnviroment,
  SdkPlatform,
  Collector
} from 'jw-trigger';

// 配置 SDK
Trigger.platform = SdkPlatform.Vue;      // 设置 SDK 引入平台为 VUE
Trigger.enviroment = SdkEnviroment.TEST; // 设置 SDK 运行环境为测试环境
Trigger.logLevel = "DEBUG";              // 设置 SDK 内部日志等级为 DEBUG 级别
Vue.use(Trigger, {                       // 安装 SDK
  router: router
})

// 自定义埋点、埋点内容为对象时可先使用 JSON.stringify() 序列化
Collector.record('自定义埋点字符串')

Vue.config.productionTip = false

new Vue({
  router,
  render: h => h(App)
}).$mount('#app')
